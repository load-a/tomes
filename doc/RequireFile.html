def dig_up_files(directory, files = [])
	header = directory + '/'

	files << list_file_paths(directory)
	folders = list_directories(directory)

	return files if folders.empty?
		
	folders.map {|folder| dig_up_files(header+folder, files.flatten) }.flatten.uniq
end

def list_file_paths(directory)
	$gtk.list_files(directory).select { |file| file.include? '.rb' }
														.map { |file| directory + '/' + file }
														.flatten
end

Requires all .rb files in a given directory. Files can be ignored via the ignore parameter or by including them in a folder with a dot in its name.

Parameters:

  • directory (String) (defaults to: 'app') —

    The directory that contains the files to be required. Defaults to ‘app’.

  • ignore (Array<String>) (defaults to: ['app/main.rb']) —

    A list of all the files you want ignored. Defaults to ‘app/main.rb’ @note The full file path (starting with ‘app/’) must be specified.

def require_all_ruby_files(directory: 'app', ignore: ['app/main.rb'])
	dig_up_files(directory).reject { |file| ignore.include? file }
													.each {|file| require file}
end

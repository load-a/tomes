def calculate_current_time

  self.minutes = seconds_to_minutes(seconds)
  self.hours = minutes_to_hours(minutes)

def formatted_time_table
  calculate_current_time
  {
    frames: frames % 60,
    seconds: seconds % 60,
    minutes: minutes % 60,
    hours: hours
  }
end

def full_status
  calculate_current_time
  {
    name: name,
    frames: frames,
    seconds: seconds.round(2),
    minutes: minutes.round(2),
    hours: hours.floor,
    on: on,
    state: state
  }
end

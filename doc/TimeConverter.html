def convert_time_to_seconds(hours: 0, minutes: 0, seconds: 0, frames: 0)
	(hours * 60**2) + (minutes * 60) + seconds + (frames / ASSUMED_FRAMES_PER_SECOND)
end

def formatted_time_table(hours: 0, minutes: 0, seconds: 0, frames: 0)
	{
		frames: frames%60,
		seconds: seconds%60,
		minutes: minutes%60,
		hours: hours
	}
end
